telco_cleaning.sql
SELECT * FROM "Telco-Customer-Churn" ;
SELECT COUNT(*) FROM "Telco-Customer-Churn";
SELECT Churn, COUNT(*) AS total_customers
FROM "Telco-Customer-Churn"
GROUP BY Churn;
SELECT Churn, AVG(tenure) AS avg_tenure
FROM "Telco-Customer-Churn"
GROUP BY Churn;
SELECT Churn, AVG(MonthlyCharges) AS avg_monthly_charges
FROM "Telco-Customer-Churn"
GROUP BY Churn;
SELECT Contract, COUNT(*) AS total
FROM "Telco-Customer-Churn"
GROUP BY Contract
ORDER BY total DESC;
SELECT *
FROM "Telco-Customer-Churn"
WHERE TRIM(TotalCharges) = '';
DELETE FROM "Telco-Customer-Churn"
WHERE TRIM(TotalCharges) = '';
SELECT * FROM "Telco-Customer-Churn" ;
SELECT COUNT(*) AS total_rows,
SUM(CASE WHEN TRIM(TotalCharges) = '' THEN 1 ELSE 0 END) AS blanks_remaining
FROM "Telco-Customer-Churn";
CREATE TABLE telco_clean AS
SELECT *,
CAST(TotalCharges AS FLOAT) AS TotalCharges_float
FROM "Telco-Customer-Churn";
-- Check first 5 rows
SELECT * FROM telco_clean LIMIT 5;
-- Ensure TotalCharges_float is numeric
SELECT AVG(TotalCharges_float), MIN(TotalCharges_float), MAX(TotalCharges_float)
FROM telco_clean;
SELECT Churn, COUNT(*) AS total_customers
FROM telco_clean
GROUP BY Churn;
SELECT Churn, AVG(MonthlyCharges) AS avg_monthly_charges
FROM telco_clean
GROUP BY Churn;
SELECT Contract, AVG(tenure) AS avg_tenure
FROM telco_clean
GROUP BY Contract;
SELECT InternetService,
SUM(CASE WHEN Churn='Yes' THEN 1 ELSE 0 END)*100.0/COUNT(*) AS churn_rate
FROM telco_clean
GROUP BY InternetService;
SELECT * FROM telco_clean
